1. Array Operations C

Write a C program to perform various operations on arrays (insertion, deletion, searching, sorting).

#include <stdio.h>

void display(int arr[], int n) {
    for(int i=0; i<n; i++) printf("%d ", arr[i]);
    printf("\n");
}

void insert(int arr[], int *n, int pos, int val) {
    for(int i=*n; i>pos; i--) arr[i] = arr[i-1];
    arr[pos] = val;
    (*n)++;
}

int main() {
    int arr[50] = {1, 2, 3, 4, 5};
    int n = 5;
    
    // Insert at position 2
    insert(arr, &n, 2, 10);
    display(arr, n);
    
    return 0;
}
2. File Handling C

Write a C program to read from a file, count vowels and write the count to another file.

#include <stdio.h>
#include <ctype.h>

int main() {
    FILE *fin, *fout;
    char ch;
    int count = 0;
    
    fin = fopen("input.txt", "r");
    fout = fopen("output.txt", "w");
    
    if(fin == NULL || fout == NULL) {
        printf("File error!");
        return 1;
    }
    
    while((ch = fgetc(fin)) != EOF) {
        ch = tolower(ch);
        if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')
            count++;
    }
    
    fprintf(fout, "Vowel count: %d", count);
    
    fclose(fin);
    fclose(fout);
    return 0;
}

5. File Handling Python

Write a Python program to read a CSV file and calculate average of a column.

import csv

def calculate_average(filename, column):
    total = 0
    count = 0
    
    with open(filename, 'r') as file:
        reader = csv.DictReader(file)
        for row in reader:
            try:
                total += float(row[column])
                count += 1
            except ValueError:
                continue
                
    return total / count if count > 0 else 0

# Example usage
average = calculate_average('data.csv', 'price')
print(f"Average price: {average:.2f}")

7. Binary Search Tree DSA

Implement Binary Search Tree with insertion, deletion and traversal operations.

class Node {
    int key;
    Node left, right;
    
    public Node(int item) {
        key = item;
        left = right = null;
    }
}

class BST {
    Node root;
    
    BST() { root = null; }
    
    void insert(int key) { root = insertRec(root, key); }
    
    Node insertRec(Node root, int key) {
        if(root == null) {
            root = new Node(key);
            return root;
        }
        
        if(key < root.key)
            root.left = insertRec(root.left, key);
        else if(key > root.key)
            root.right = insertRec(root.right, key);
            
        return root;
    }
    
    void inorder() { inorderRec(root); }
    
    void inorderRec(Node root) {
        if(root != null) {
            inorderRec(root.left);
            System.out.print(root.key + " ");
            inorderRec(root.right);
        }
    }
}

public class Main {
    public static void main(String[] args) {
        BST tree = new BST();
        tree.insert(50);
        tree.insert(30);
        tree.insert(20);
        tree.insert(40);
        tree.insert(70);
        tree.insert(60);
        tree.insert(80);
        
        tree.inorder(); // Prints in sorted order
    }
}
class Node:
    def __init__(self, key):
        self.left = None
        self.right = None
        self.val = key

class BST:
    def __init__(self):
        self.root = None
    
    def insert(self, key):
        if self.root is None:
            self.root = Node(key)
        else:
            self._insert(self.root, key)
    
    def _insert(self, node, key):
        if key < node.val:
            if node.left is None:
                node.left = Node(key)
            else:
                self._insert(node.left, key)
        else:
            if node.right is None:
                node.right = Node(key)
            else:
                self._insert(node.right, key)
    
    def inorder(self):
        return self._inorder(self.root, [])
    
    def _inorder(self, node, result):
        if node:
            self._inorder(node.left, result)
            result.append(node.val)
            self._inorder(node.right, result)
        return result

8. Dijkstra's Algorithm DAA

Implement Dijkstra's algorithm to find shortest paths in a graph.

import java.util.*;

class Graph {
    private int V;
    private List<List<Node>> adj;
    
    Graph(int v) {
        V = v;
        adj = new ArrayList<>(v);
        for(int i=0; i<v; i++)
            adj.add(new ArrayList<>());
    }
    
    void addEdge(int u, int v, int weight) {
        adj.get(u).add(new Node(v, weight));
        adj.get(v).add(new Node(u, weight)); // Undirected graph
    }
    
    void dijkstra(int src) {
        int[] dist = new int[V];
        Arrays.fill(dist, Integer.MAX_VALUE);
        dist[src] = 0;
        
        PriorityQueue<Node> pq = new PriorityQueue<>(V, Comparator.comparingInt(a -> a.weight));
        pq.add(new Node(src, 0));
        
        while(!pq.isEmpty()) {
            Node node = pq.poll();
            int u = node.vertex;
            
            for(Node neighbor : adj.get(u)) {
                int v = neighbor.vertex;
                int weight = neighbor.weight;
                
                if(dist[v] > dist[u] + weight) {
                    dist[v] = dist[u] + weight;
                    pq.add(new Node(v, dist[v]));
                }
            }
        }
        
        System.out.println("Vertex Distance from Source");
        for(int i=0; i<V; i++)
            System.out.println(i + "\t\t" + dist[i]);
    }
    
    class Node {
        int vertex, weight;
        Node(int v, int w) { vertex = v; weight = w; }
    }
}

public class Main {
    public static void main(String[] args) {
        Graph g = new Graph(5);
        g.addEdge(0, 1, 4);
        g.addEdge(0, 2, 8);
        g.addEdge(1, 2, 2);
        g.addEdge(1, 3, 5);
        g.addEdge(2, 3, 5);
        g.addEdge(2, 4, 9);
        g.addEdge(3, 4, 4);
        
        g.dijkstra(0);
    }
}
import heapq

class Graph:
    def __init__(self, vertices):
        self.V = vertices
        self.graph = [[] for _ in range(vertices)]
    
    def add_edge(self, u, v, w):
        self.graph[u].append((v, w))
        self.graph[v].append((u, w))  # Undirected graph
    
    def dijkstra(self, src):
        dist = [float('inf')] * self.V
        dist[src] = 0
        heap = [(0, src)]
        
        while heap:
            current_dist, u = heapq.heappop(heap)
            
            if current_dist > dist[u]:
                continue
                
            for v, weight in self.graph[u]:
                if dist[v] > dist[u] + weight:
                    dist[v] = dist[u] + weight
                    heapq.heappush(heap, (dist[v], v))
        
        print("Vertex Distance from Source")
        for i in range(self.V):
            print(f"{i}\t\t{dist[i]}")

9. FIR Filter DSP

Implement a Finite Impulse Response (FIR) filter in Python.

import numpy as np
import matplotlib.pyplot as plt

def fir_filter(input_signal, coefficients):
    output = np.zeros_like(input_signal)
    n_coeff = len(coefficients)
    
    for i in range(len(input_signal)):
        for j in range(n_coeff):
            if i - j >= 0:
                output[i] += coefficients[j] * input_signal[i - j]
    
    return output

# Example usage
fs = 1000  # Sampling frequency
t = np.arange(0, 1, 1/fs)
f1, f2 = 50, 250  # Signal frequencies

# Create a signal with two frequencies
signal = np.sin(2 * np.pi * f1 * t) + 0.5 * np.sin(2 * np.pi * f2 * t)

# Design a low-pass FIR filter to remove the higher frequency
cutoff = 100  # Cutoff frequency
nyquist = 0.5 * fs
normal_cutoff = cutoff / nyquist
numtaps = 101  # Number of filter coefficients

# Create filter coefficients (Hamming window)
taps = np.sinc(2 * normal_cutoff * (np.arange(numtaps) - (numtaps - 1) / 2))
taps *= np.hamming(numtaps)
taps /= np.sum(taps)

# Apply the filter
filtered_signal = fir_filter(signal, taps)

# Plot results
plt.figure(figsize=(10, 6))
plt.plot(t, signal, label='Original Signal')
plt.plot(t, filtered_signal, label='Filtered Signal')
plt.xlabel('Time [s]')
plt.ylabel('Amplitude')
plt.legend()
plt.grid()
plt.show()
